### 1. Storage device setup

# sdb is the low speed blockdevice
# sdc is the high speed blockdevice

# 1.a. with thin-provisionning (recommended)
sudo apt-get install thin-provisioning-tools
sudo pvcreate /dev/sdb
sudo vgcreate vg-archive /dev/sdb
sudo lvcreate -l 100%FREE -T vg-archive/thinpool -V100G -n lv0 --errorwhenfull y
sudo mkfs -t ext4 -L MAILARCHIVE -E nodiscard /dev/vg-archive/lv-archive
sudo vgcreate vg-fast /dev/sdc
sudo lvcreate -l 100%FREE -T vg-fast/thinpool -V100G -n lv0 --errorwhenfull y
sudo mkfs -t ext4 -L MAILCURRENT -E nodiscard /dev/vg-fast/lv-fast



# 1.b. thin-provisionning with automatic resize (not used in our case because we need to manually extend the blockdevice)
sudo apt-get install thin-provisioning-tools
sudo vi /etc/lvm/lvm.conf
        thin_pool_autoextend_threshold = 70
	thin_pool_autoextend_percent = 20
sudo pvcreate /dev/sdb
sudo vgcreate vg /dev/sdb
sudo lvcreate -L 5G -T vg/thinpool -V100G -n lv0
sudo mkfs -t ext4 -L MAILARCHIVE -E nodiscard /dev/vg/lv0


# 1.c. no thin-provisionning (not recommended)
sudo pvcreate /dev/sdb
sudo vgcreate vg /dev/sdb
sudo lvcreate -n lv0 -l 100%FREE vg
sudo mkfs -t ext4 -L MAILARCHIVE /dev/vg/lv0



### 2. Mount
sudo mkdir -p /mnt/mailarchive
echo 'LABEL=MAILARCHIVE	/mnt/mailarchive	ext4	defaults	0 2' >> /etc/fstab
sudo mount /mnt/mailarchive

sudo mkdir -p /mnt/mailcurrent
echo 'LABEL=MAILCURRENT	/mnt/mailcurrent	ext4	defaults	0 2' >> /etc/fstab
sudo mount /mnt/mailcurrent

### 6. FAQ

Q: `pvcreate`command returns 'Device /dev/sdx excluded by a filter' error.
A: Run with -vvv to get more details. In my case: 
        /dev/sdc: Skipping: Partition table signature found
   We understand the device was previously partionned. Fix by wiping data: `sudo dd if=/dev/zero of=/dev/sdc bs=1M count=1`
